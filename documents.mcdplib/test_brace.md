\begin{defn}[Upper closure]
The operator~$\uparrow$ maps a subset to the smallest upper set that
includes it:
\begin{eqnarray*}
\uparrow\colon\mathcal{P}(P) & \rightarrow & UP,\\
S & \mapsto & \{y\in P:\exists\,x\in S:x\leq y\}.
\end{eqnarray*}
\end{defn}
